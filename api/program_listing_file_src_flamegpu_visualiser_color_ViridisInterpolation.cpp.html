#include "flamegpu/visualiser/color/ViridisInterpolation.h"

#include <sstream>

ViridisInterpolation::ViridisInterpolation(const std::string& _variable_name, const float& _min_bound, const float& _max_bound)
    : min_bound(_min_bound)
    , max_bound(_max_bound)
    , variable_name(_variable_name)
    , invert_palette(false) {
    if (min_bound > max_bound) {
        invert_palette = true;
        std::swap(min_bound, max_bound);
    }
}
ViridisInterpolation& ViridisInterpolation::setBounds(const float& _min_bound, const float& _max_bound) {
    min_bound = _min_bound;
    max_bound = _max_bound;
    if (min_bound > max_bound) {
        invert_palette = true;
        std::swap(min_bound, max_bound);
    }
    return *this;
}

std::string ViridisInterpolation::getSrc() const {
    static const std::array<const Color, 256> &raw_colors = rawColors();
    std::stringstream ss;
    ss << "uniform samplerBuffer color_arg;" << "\n";
    // Create a function that maps an index to a color
    // A switch should work, unsure if I can declare an actual array in shaders without using buffers
    ss << "vec3 getRawColor(int i) {" << "\n";
    ss << "    switch (i) {" << "\n";
    for (size_t i = 0; i < raw_colors.size(); ++i) {
        ss << "      case " << i << ":" << "\n";
        ss << "        return vec3(" << raw_colors[i].r << ", " << raw_colors[i].g << ", " << raw_colors[i].b << ");" << "\n";
    }
    ss << "      default: // This should never be hit" << "\n";
    ss << "        return vec3(1, 1, 1);" << "\n";
    ss << "    }" << "\n";
    ss << "}" << "\n";
    ss << "vec4 calculateColor() {" << "\n";
    // Fetch the modifier from texture cache
    ss << "    float modifier = texelFetch(color_arg, gl_InstanceID).x;" << "\n";
    // Clamp the modifier to bounds
    ss << "    modifier = clamp(modifier, " << min_bound << ", " << max_bound << ");" << "\n";
    // Scale modifier to range [0.0, 255.0]
    ss << "    modifier = 255 * (modifier - " << min_bound << ") / " << (max_bound - min_bound) << ";" << "\n";
    // Decide which colors to pull, and how to lerp
    ss << "    const float a = floor(modifier);" << "\n";
    ss << "    const float t = modifier - a;" << "\n";
    // Optionally invert the palette
    ss << "    const vec3 c0 = getRawColor(" << (invert_palette ? "255 - " : "") << "int(a));" << "\n";
    ss << "    const vec3 c1 = getRawColor(" << (invert_palette ? "255 - " : "") << "int(ceil(modifier)));" << "\n";
    // Lerp and return the chosen color (alpha = 1.0f)
    ss << "    return vec4(c0 * (1.0 - t) + c1 * t, 1.0);" << "\n";
    ss << "}" << "\n";
    return ss.str();
}
std::string ViridisInterpolation::getSamplerName() const {
    return "color_arg";
}
std::string ViridisInterpolation::getAgentVariableName() const {
    return variable_name;
}
